# ==========================================
# 智能聊天助手 - 环境及依赖配置文件
# ==========================================
# [操作系统环境]: 推荐 Windows 10/11 (也可以运行在 Linux / macOS)
# [Python环境]: 必须使用 Python 3.10.x（开发及测试环境为 Python 3.10.10）
# ==========================================

# 1. 核心 Web 服务框架及其 ASGI 服务器
fastapi==0.104.1      # 用于构建高性能 HTTP API 端点
uvicorn==0.24.0       # ASGI 服务器，用于运行 FastAPI 服务

# 2. 数据存储相关驱动
mysql-connector-python==8.2.0  # MySQL 官方 Python 驱动，用于连接本地或远程 MySQL 进行对话记录的持久化
redis==5.0.1          # Redis Python 客户端，负责连接外部 Redis Server 提供多进程可共享的高速缓存

# 3. 外部 API 交互
requests==2.31.0      # 高级 HTTP 客户端库，本系统用于向后端的 Ollama 服务发起 REST API 调用 (同步及流式)

# 4. 其他附属与被依赖工具
python-multipart==0.0.6 # 处理 form-data 和 multipart 的 FastAPI 依赖项，用于以防将来的表单请求
cachetools==5.3.3     # 可选的内存缓存辅助库（当前代码内部基于 dict 实现了类似算法，此库保留作为未来复杂内存缓存架构或者已有环境的依赖保留）
# 5. Python 内置标准库 (随 Python 3.10.x 自动安装，无需 pip install)
